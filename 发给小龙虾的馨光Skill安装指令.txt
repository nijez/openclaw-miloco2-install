```bash
TARGET=/tmp/install-xinguang-skill.sh
LOG=/tmp/xinguang-skill-install-current.log
STATE=/tmp/xinguang-skill-install.state
PIDFILE=/tmp/xinguang-skill-install.pid
EXPECTED_VERSION=2026-06-26.2

if [ -f "$PIDFILE" ]; then
  OLD_PID="$(cat "$PIDFILE" 2>/dev/null || true)"
  if [ -n "$OLD_PID" ] && ps -p "$OLD_PID" >/dev/null 2>&1; then
    echo "馨光 Skill 安装任务已经在运行，请不要重复启动。"
    echo "如需查看进度，请发送：查看馨光 Skill 安装进度"
    exit 0
  fi
fi

rm -f "$TARGET"
for URL in \
  https://nijez.github.io/xingguang-ai-lighting-guide/install-xinguang-skill.sh \
  https://raw.githubusercontent.com/nijez/xingguang-ai-lighting-guide/main/install-xinguang-skill.sh \
  https://cdn.jsdelivr.net/gh/nijez/xingguang-ai-lighting-guide@main/install-xinguang-skill.sh
do
  echo "尝试下载馨光 Skill 安装脚本: $URL"
  if curl -fsSL "$URL" -o "$TARGET" && grep -q "XINGUANG_SKILL_INSTALLER_VERSION=\"$EXPECTED_VERSION\"" "$TARGET"; then
    echo "馨光 Skill 安装脚本下载成功，版本 $EXPECTED_VERSION"
    break
  fi
  echo "当前源失败或版本不匹配，继续尝试下一个源"
done

grep -q "XINGUANG_SKILL_INSTALLER_VERSION=\"$EXPECTED_VERSION\"" "$TARGET" || { echo "馨光 Skill 安装脚本下载失败或版本不正确"; exit 1; }
chmod +x "$TARGET"

nohup env \
  XINGUANG_TARGET_HOME="林坞店" \
  XINGUANG_TARGET_ROOM="测试" \
  XINGUANG_TARGET_DEVICE_NAME="软膜" \
  XINGUANG_TARGET_PRODUCT="馨光RGBCW幻彩灯带" \
  INSTALL_ACTION=continue \
  INSTALL_NONINTERACTIVE=1 \
  bash "$TARGET" >> "$LOG" 2>&1 &
echo $! > "$PIDFILE"
sleep 2

echo "馨光 Skill 后台安装已启动"
echo "安装脚本版本: $EXPECTED_VERSION"
echo "日志文件: $LOG"
echo "状态文件: $STATE"
echo "说明: 安装任务已在服务器后台运行。请稍后发送「查看馨光 Skill 安装进度」。不要重复发送本安装指令。"
```
